```javascript
/**
 * Definition for singly-linked list.
 * function ListNode(val) {
 *     this.val = val;
 *     this.next = null;
 * }
 */
/**
 * @param {ListNode} head
 * @return {number}
 */
var getDecimalValue = function(head) {
    var res = '';
    while(head!=null){
        res = res + head.val;
        head = head.next;
    }
    var sum = 0,cur=1;
    for(var i = res.length-1;i>=0;i--){
        sum = sum + res[i] * cur;
        cur = cur * 2;
    }
    return sum;
};
```
